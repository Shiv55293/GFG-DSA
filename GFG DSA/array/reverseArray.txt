class Solution {   
  static int[] reverse(int arr[])
  {
    int low=0,high=arr.length-1;
    while(low<high)
      {
        int temp= arr[low];
        arr[low]= arr[high];
        arr[high]= temp;
      }
    return arr;
  }
  public static void main(String[] args) {
    int arr[]= {10,30,40,50,20};
       arr=reverse(arr);
         System.out.println(Arrays.toString(arr));
  }
}
